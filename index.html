precision mediump float; precision mediump int; varying vec2 _uv; uniform float time; uniform sampler2D backbuffer; uniform int plane_id; void main() { vec2 uv = _uv; vec3 color = vec3(0.); float grid = 0.; grid += clamp(sin(uv.x * 200. + time * 10.) - 0.5, 0., 1.); grid += clamp(cos(uv.y * 200. + sin(time) * 10.) - 0.5, 0., 1.); vec3 grid_color = vec3(0.); if(plane_id == 0) { grid_color.r = 1.; } else if(plane_id == 1) { grid_color.g = 1.; } else if(plane_id == 2) { grid_color.b = 1.; } color += grid_color * grid; color += texture(backbuffer, uv).yyz * 0.95; gl_FragColor = vec4(color, 1.); }
